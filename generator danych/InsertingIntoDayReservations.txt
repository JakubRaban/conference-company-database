import java.util.Random;

public class Shuffler {

    public static void main(String ... args) {

        Random r = new Random();

        int numberOfReservations = 948;
        int minDayId = 36;
        int maxDayID = 264;
        int adultSeats;
        int studentSeats;

        for(int i = 0; i < 8000; i++) {
            adultSeats = r.nextInt(20);
            studentSeats = r.nextInt(20);
            int dayId = r.nextInt(maxDayID - minDayId) + 36;
            int reservationId = r.nextInt(numberOfReservations);
            StringBuilder sql = new StringBuilder("insert into ConferenceDayReservation " +
                    "(ConferenceDayID, ReservedAdultSeats, ReservedStudentSeats, ReservationID) values (");
            if (r.nextDouble() > 2.0 / 3.0) {
                if(r.nextDouble() < 1.0 / 2.0) {
                    studentSeats = 0; adultSeats = 1;
                } else {
                    studentSeats = 1; adultSeats = 0;
                }
            }
            sql.append(dayId + ", " + adultSeats + ", " + studentSeats + ", " + reservationId + ")");
            sql.append("\ngo");
            System.out.println(sql);
        }

    }

}
